**1. Create favorites provider (`ingredient_favorites_provider.dart`):**

```dart
@riverpod
class IngredientFavorites extends _$IngredientFavorites {
  @override
  FutureOr<List<Ingredient>> build() async {
    final repo = ref.watch(ingredientRepositoryProvider);
    return repo.getFavorites();
  }

  Future<void> toggleFavorite(String ingredientId) async {
    final repo = ref.read(ingredientRepositoryProvider);

    // Optimistic update: toggle in local state immediately
    state = AsyncData(
      state.valueOrNull?.map((i) {
        if (i.id == ingredientId) {
          return i.copyWith(isFavorite: !i.isFavorite);
        }
        return i;
      }).toList() ?? [],
    );

    // Persist to Drift (syncStatus set to pending for future Supabase sync)
    await repo.toggleFavorite(ingredientId);

    // Refresh from source of truth
    if (!ref.mounted) return;
    state = AsyncData(await repo.getFavorites());
  }
}
```

**2. Create selected-today provider (`selected_today_provider.dart`):**

Follow RESEARCH Pattern 4. This is a `@Riverpod(keepAlive: true)` provider — it must survive navigation because Phase 4 recipe discovery reads it.

```dart
@Riverpod(keepAlive: true)
class SelectedTodayNotifier extends _$SelectedTodayNotifier {
  @override
  FutureOr<Set<String>> build() async {
    final repo = ref.watch(ingredientRepositoryProvider);
    final user = ref.watch(currentUserProvider);
    if (user == null) return {};
    final ids = await repo.getSelectedToday(user.id);
    return ids.toSet();
  }

  Future<void> toggle(String ingredientId) async {
    final repo = ref.read(ingredientRepositoryProvider);
    final user = ref.read(currentUserProvider);
    if (user == null) return;

    final current = state.valueOrNull ?? {};
    final updated = Set<String>.from(current);

    if (updated.contains(ingredientId)) {
      updated.remove(ingredientId);
      await repo.removeSelectedToday(ingredientId);
    } else {
      updated.add(ingredientId);
      await repo.addSelectedToday(ingredientId, user.id);
    }

    if (!ref.mounted) return;
    state = AsyncData(updated);
  }

  Future<void> clearAll() async {
    final repo = ref.read(ingredientRepositoryProvider);
    final user = ref.read(currentUserProvider);
    if (user == null) return;
    await repo.clearSelectedToday(user.id);
    if (!ref.mounted) return;
    state = const AsyncData({});
  }

  Set<String> get selectedIds => state.valueOrNull ?? {};
  int get count => selectedIds.length;
}
```

**CRITICAL:** Use `@Riverpod(keepAlive: true)` — NOT auto-dispose. This state must persist across screen navigation so Phase 4 recipe discovery can read it from any route. Check `ref.mounted` after every `await` (Riverpod 3 requirement).
