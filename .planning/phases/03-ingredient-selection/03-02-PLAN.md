**1. Create debounced search provider (`ingredient_search_provider.dart`):**

Follow RESEARCH Pattern 1 exactly. Use `@riverpod` annotation (Riverpod 3 code-gen):
```dart
@riverpod
class IngredientSearch extends _$IngredientSearch {
  Timer? _debounce;

  @override
  FutureOr<List<String>> build() => [];

  Future<void> search(String query) async {
    _debounce?.cancel();
    if (query.length < 2) {
      state = const AsyncData([]);
      return;
    }
    _debounce = Timer(const Duration(milliseconds: 300), () async {
      state = const AsyncLoading();
      state = await AsyncValue.guard(() =>
        ref.read(ingredientRepositoryProvider).searchSuggestions(query),
      );
    });
    ref.onDispose(() => _debounce?.cancel());
  }
}
```

**3. Create filter provider (`ingredient_filter_provider.dart`):**
```dart
@riverpod
class IngredientFilterNotifier extends _$IngredientFilterNotifier {
  @override
  Set<DietaryRestriction> build() => {};

  void toggle(DietaryRestriction restriction) {
    final updated = Set<DietaryRestriction>.from(state);
    if (updated.contains(restriction)) {
      updated.remove(restriction);
    } else {
      updated.add(restriction);
    }
    state = updated;
  }

  void clear() => state = {};
}

@riverpod
Future<List<Ingredient>> filteredIngredients(Ref ref) async {
  final filters = ref.watch(ingredientFilterNotifierProvider);
  if (filters.isEmpty) return [];
  final repo = ref.watch(ingredientRepositoryProvider);
  return repo.filterByDietary(filters);
}
```
