**1. Create `template_notifier.dart`:**
```dart
@riverpod
class TemplateNotifier extends _$TemplateNotifier {
  @override
  Stream<List<PlanTemplate>> build() {
    final repo = ref.watch(templateRepositoryProvider);
    final userId = ref.watch(currentUserIdProvider);
    return repo.watchTemplates(userId);
  }

  Future<void> saveCurrentWeek({required String name, required DateTime weekStart}) async {
    final repo = ref.read(templateRepositoryProvider);
    final userId = ref.read(currentUserIdProvider);
    await repo.saveCurrentWeek(name: name, userId: userId, weekStart: weekStart);
  }

  Future<void> loadTemplate({required String templateId, required DateTime targetWeekStart}) async {
    final repo = ref.read(templateRepositoryProvider);
    final userId = ref.read(currentUserIdProvider);
    await repo.loadTemplate(templateId: templateId, userId: userId, targetWeekStart: targetWeekStart);
  }

  Future<void> deleteTemplate(String templateId) async {
    final repo = ref.read(templateRepositoryProvider);
    await repo.deleteTemplate(templateId);
  }
}
```

Create a `templateRepositoryProvider` that instantiates TemplateRepository with the AppDatabase.
